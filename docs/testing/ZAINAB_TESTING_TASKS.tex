\section{Accessibility Testing Tasks}

\subsection{Task 9: WCAG 2.1 AA Compliance Testing}

\textbf{Objective:} Ensure application meets accessibility standards (NFR-006)

\textbf{Tools:}
\begin{itemize}
    \item axe DevTools (browser extension)
    \item WAVE (browser extension)
    \item Lighthouse (Chrome DevTools)
    \item Screen reader (NVDA or VoiceOver)
\end{itemize}

\textbf{Test Cases:}

\begin{enumerate}
    \item \textbf{Automated Accessibility Audit}
    \begin{itemize}
        \item Install axe DevTools extension
        \item Navigate to each page:
        \begin{itemize}
            \item Home page
            \item Login page
            \item Dashboard
            \item Profile page
            \item Video call page
            \item Forum page
            \item Glossary page
        \end{itemize}
        \item Run axe scan on each page
        \item Document any violations
        \item Take screenshots of axe results
    \end{itemize}

    \textbf{Expected Result:} Zero critical or serious violations

    \item \textbf{Color Contrast Testing}
    \begin{itemize}
        \item Use WAVE to check color contrast
        \item Verify all text meets 4.5:1 ratio (NFR-006)
        \item Test high contrast mode
        \item Check buttons and interactive elements (3:1 ratio)
    \end{itemize}

    \item \textbf{Keyboard Navigation Testing}
    \begin{itemize}
        \item Navigate entire site using only keyboard (Tab key)
        \item Verify all interactive elements are reachable
        \item Check focus indicators are visible
        \item Test Shift+Tab (reverse navigation)
        \item Verify logical tab order
        \item Test keyboard shortcuts (if any)
    \end{itemize}

    \textbf{Checklist:}
    \begin{itemize}
        \item[$\square$] Can login using keyboard only
        \item[$\square$] Can navigate menu with Tab
        \item[$\square$] Can activate buttons with Enter/Space
        \item[$\square$] Can close modals with Escape
        \item[$\square$] Focus indicators clearly visible
    \end{itemize}

    \item \textbf{Screen Reader Testing}
    \begin{itemize}
        \item Launch screen reader (NVDA or VoiceOver)
        \item Navigate through application
        \item Verify all content is announced
        \item Check ARIA labels are present
        \item Test form field labels
        \item Verify button purposes are clear
        \item Test skip navigation links
    \end{itemize}

    \textbf{Document:} Record audio or write notes on screen reader experience

    \item \textbf{Zoom \& Text Scaling}
    \begin{itemize}
        \item Test browser zoom at 200\% (NFR-006)
        \item Verify no content is cut off
        \item Check horizontal scrolling is minimal
        \item Test text-only zoom
    \end{itemize}

    \item \textbf{Captions \& Alternative Text}
    \begin{itemize}
        \item Verify all images have alt text
        \item Check video player has captions
        \item Test caption customization (font size)
        \item Verify captions appear within 3 seconds (FR-004.1)
    \end{itemize}
\end{enumerate}

\textbf{Deliverable:} Accessibility compliance report with:
\begin{itemize}
    \item axe scan results for each page
    \item Keyboard navigation test log
    \item Screen reader testing notes
    \item List of any accessibility issues found
\end{itemize}

\section{Responsive Design Testing Tasks}

\subsection{Task 12: Mobile Responsiveness Testing}

\textbf{Objective:} Verify responsive design (NFR-007)

\textbf{Screen Sizes to Test:}
\begin{itemize}
    \item Mobile: 375px × 667px (iPhone)
    \item Tablet: 768px × 1024px (iPad)
    \item Desktop: 1366px × 768px
    \item Large Desktop: 1920px × 1080px
\end{itemize}

\textbf{Test Cases:}

\begin{enumerate}
    \item \textbf{Chrome DevTools Device Emulation}
    \begin{itemize}
        \item Open Chrome DevTools (F12)
        \item Click device toolbar icon (Ctrl+Shift+M)
        \item Select different device presets
        \item Verify layout adapts correctly
    \end{itemize}

    \item \textbf{Elements to Check:}
    \begin{itemize}
        \item Navigation menu collapses to hamburger on mobile
        \item Buttons are touch-friendly (44×44px minimum)
        \item Text is readable without zooming
        \item Forms are easy to fill on mobile
        \item Images scale appropriately
        \item No horizontal scrolling
    \end{itemize}

    \item \textbf{Touch Interactions}
    \begin{itemize}
        \item Test tap targets are large enough
        \item Test swipe gestures (if any)
        \item Verify no hover-only interactions
    \end{itemize}
\end{enumerate}
