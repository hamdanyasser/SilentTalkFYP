\section{Backend API Testing Tasks}

\subsection{Task 1: Authentication API Testing}

\textbf{Objective:} Verify all authentication endpoints work correctly

\textbf{Test Cases:}

\begin{enumerate}
    \item \textbf{User Registration (FR-001.1)}
    \begin{itemize}
        \item Test successful registration with valid data
        \item Test registration with existing email (should fail)
        \item Test registration with weak password (should fail)
        \item Test email verification flow
        \item Verify password is hashed in database
    \end{itemize}

    \textbf{How to Test:}
    \begin{lstlisting}[language=bash]
# Start the backend server
cd server
dotnet run

# Use Postman or curl
curl -X POST http://localhost:5000/api/auth/register \
  -H "Content-Type: application/json" \
  -d '{
    "email": "yasser@test.com",
    "password": "Test123!@#",
    "displayName": "Yasser Test"
  }'
    \end{lstlisting}

    \textbf{Expected Result:}
    \begin{itemize}
        \item Status code: 201 Created
        \item Response contains user ID and JWT token
        \item User record exists in database
    \end{itemize}

    \item \textbf{User Login (FR-001.2, FR-001.3)}
    \begin{itemize}
        \item Test successful login with valid credentials
        \item Test login with incorrect password (should fail)
        \item Test login with non-existent email (should fail)
        \item Verify JWT token is returned
        \item Test account lockout after 5 failed attempts (FR-001.9)
    \end{itemize}

    \item \textbf{JWT Token Validation (FR-001.3)}
    \begin{itemize}
        \item Test protected endpoints with valid token
        \item Test protected endpoints without token (should return 401)
        \item Test protected endpoints with expired token
        \item Test token refresh mechanism
    \end{itemize}

    \item \textbf{Password Reset (FR-001.8)}
    \begin{itemize}
        \item Request password reset
        \item Verify reset token generation
        \item Test password reset with valid token
        \item Test password reset with expired token
    \end{itemize}
\end{enumerate}

\textbf{Deliverable:} Complete test report with screenshots for each test case

\section{ML Service Testing Tasks}

\subsection{Task 4: Sign Recognition API Testing}

\textbf{Objective:} Validate ML model accuracy and performance (FR-002)

\textbf{Test Cases:}

\begin{enumerate}
    \item \textbf{Model Health Check}
    \begin{lstlisting}[language=bash]
curl http://localhost:8000/health
    \end{lstlisting}

    \textbf{Expected:} Model accuracy >= 85\% (NFR-001)

    \item \textbf{Sign Recognition Request (FR-002.1, FR-002.4)}
    \begin{lstlisting}[language=bash]
curl -X POST http://localhost:8000/api/ml/recognize \
  -H "Content-Type: application/json" \
  -d '{
    "landmarks": [...],
    "handedness": "Right"
  }'
    \end{lstlisting}

    \textbf{Verify:}
    \begin{itemize}
        \item Response time < 100ms (NFR-001)
        \item Confidence score provided (FR-002.7)
        \item Correct sign predicted
    \end{itemize}

    \item \textbf{Performance Testing (FR-002.3, FR-002.4)}
    \begin{itemize}
        \item Test with 100 consecutive requests
        \item Measure average inference time
        \item Verify all requests complete within 100ms
        \item Test concurrent requests (simulate 10 users)
    \end{itemize}

    \item \textbf{Multi-Language Support (FR-002.2)}
    \begin{itemize}
        \item Test ASL sign recognition
        \item Test BSL sign recognition
        \item Test ISL sign recognition
        \item Verify language switching
    \end{itemize}
\end{enumerate}

\subsection{Task 5: ML Model Accuracy Validation}

\textbf{Objective:} Confirm model meets 85\% accuracy requirement

\textbf{Steps:}

\begin{enumerate}
    \item Prepare test dataset (at least 100 samples)
    \item Run predictions on test set
    \item Calculate accuracy: $\text{Accuracy} = \frac{\text{Correct Predictions}}{\text{Total Predictions}} \times 100$
    \item Document results in spreadsheet
    \item Generate confusion matrix
\end{enumerate}

\textbf{Deliverable:} Excel sheet with test results showing >= 85\% accuracy

\section{Security Testing Tasks}

\subsection{Task 7: Authentication Security Testing}

\textbf{Objective:} Verify security requirements (NFR-004)

\textbf{Test Cases:}

\begin{enumerate}
    \item \textbf{Password Security (FR-001.2, FR-001.7)}
    \begin{itemize}
        \item Verify passwords are hashed (not plaintext in DB)
        \item Test password complexity requirements
        \item Test SQL injection in login form
    \end{itemize}

    \item \textbf{JWT Security}
    \begin{itemize}
        \item Attempt to access API with tampered token
        \item Test token expiration (30 minutes)
        \item Verify refresh token mechanism
    \end{itemize}

    \item \textbf{Rate Limiting (NFR-004)}
    \begin{itemize}
        \item Make 101+ requests in 1 minute
        \item Verify request is blocked (429 status)
    \end{itemize}

    \item \textbf{SQL Injection Testing}
    \begin{itemize}
        \item Test login with SQL injection payloads
        \item Test search endpoints with malicious input
        \item Verify all inputs are sanitized
    \end{itemize}
\end{enumerate}
